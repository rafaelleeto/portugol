#include <stdio.h>

int main() {
int num, i;
int isPrimo;

for (num = 1; num <= 100; num++) {
if (num < 2) {
printf("%d não é primo\n", num);
continue;
}

isPrimo = 1; // Supondo que o número é primo

for (i = 2; i * i <= num; i++) {
if (num % i == 0) {
isPrimo = 0;
break;
}
}

if (isPrimo) {
printf("%d é primo\n", num);
} else {
printf("%d não é primo\n", num);
}
}

return 0;
}
